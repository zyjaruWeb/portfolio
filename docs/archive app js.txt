function objectDisplayFunction (projectItems) {
    let projectsDisplay = projectItems.map(function(item) {
        //create variable which holds array with available on call specified elements of each object
        //project item defines an empty variable
        //item is dummy item which then will be replaced with whatever is inserted through the function
        
        return ` 
        <article class="menuItemClass">
            <a href="${item.link}"><img src="${item.img}" class="photoClass" alt=${item.title}></a>
            <div class="itemInfoClass">
                <header>
                    <h4>${item.title}</h4>
                </header>
                <div id="underlineMenu"></div>
                <p class="itemTextClass">${item.description}</p>
            </div>
        </article>
        `
    })
    projectsDisplay = projectsDisplay.join("") //join creates array based, above html * no of turns
